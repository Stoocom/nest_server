export const uploadFile = (authClient) => {
    console.log('uploadFile');

        const drive = google.drive({ version: 'v3', auth: authClient});

        // const { file } = req;
 
        let fileMetaData = {
            name: "",
            parents: ["1hIayFYHQg6ZnousgK34l-wu1W6z5U-9B"]
        }

        drive.files.create({
            media: {
                mimeType: "text/plain",
                body: createReadStream("text.txt")

            requestBody: fileMetaData,
            fields: "id,name"
        }, (error, file) => {
            if (error) {
                return req(error)
            }
            res(file)
        })
